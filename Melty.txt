import React, { useState, useEffect, useRef, useCallback, useMemo } from 'react';
import { Plus, X, Heart, Sparkles, Moon, MoreHorizontal, Wand2, Loader2, Quote, Feather, List, BookOpen, Bookmark, Trash2, Calendar, Eye, Wind, MessageCircle, Send, ArrowRight, Lightbulb, Eraser, Check } from 'lucide-react';
import { initializeApp } from "firebase/app";
import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "firebase/auth";
import { getFirestore, collection, addDoc, deleteDoc, doc, onSnapshot, query, orderBy, serverTimestamp, updateDoc } from "firebase/firestore";

// Initialize Firebase
const firebaseConfig = JSON.parse(__firebase_config);
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

// Default Aesthetic Todos (Constant)
const DEFAULT_MELTY_TODOS = [
  { id: 1, text: "夜明けの淡い光をスプーンですくう", completed: false },
  { id: 2, text: "誰にも宛てない手紙を瓶に詰める", completed: false },
  { id: 3, text: "雨音の周波数をチューニングする", completed: false },
  { id: 4, text: "古い香水の記憶を呼び覚ます", completed: false },
  { id: 5, text: "ガラスの靴で砂浜を歩く", completed: false },
  { id: 6, text: "沈黙の輪郭を指でなぞる", completed: true },

  { id: 9, text: "枯れた花束に名前をつける", completed: false },

  { id: 11, text: "壊れた時計の針を逆さに回す", completed: false },

  { id: 14, text: "星屑をポケットに集める", completed: true },

  { id: 16, text: "誰もいない遊園地でメリーゴーランドを回す", completed: false },
  { id: 17, text: "溶けたアイスクリームの色を記録する", completed: false },
  { id: 18, text: "深海魚の夢を見る", completed: true },
  { id: 19, text: "透明なインクで日記を書く", completed: false },
  { id: 20, text: "世界の終わりの静けさを愛でる", completed: false },
];

const MeltyApp = () => {
  const [activeTab, setActiveTab] = useState('list');
  const [user, setUser] = useState(null);
  
  // Data States
  const [todos, setTodos] = useState(DEFAULT_MELTY_TODOS);
  const [inputValue, setInputValue] = useState("");
  const [storyInput, setStoryInput] = useState("");
  const [generatedStory, setGeneratedStory] = useState("");
  const [savedStories, setSavedStories] = useState([]);
  const [chatInput, setChatInput] = useState("");
  const [chatMessages, setChatMessages] = useState([]);
  
  // UI States
  const [isWeaving, setIsWeaving] = useState(false);
  const [expandedStoryId, setExpandedStoryId] = useState(null);
  const [isChatting, setIsChatting] = useState(false);
  const [savedChatIds, setSavedChatIds] = useState(new Set());
  const [isLoaded, setIsLoaded] = useState(false);
  const [isRewriting, setIsRewriting] = useState(false);
  const [isGenerating, setIsGenerating] = useState(false);
  const [modalState, setModalState] = useState({ isOpen: false, text: null, type: 'oracle', loading: false });
  
  // Refs
  const chatEndRef = useRef(null);
  const longPressTimer = useRef(null);

  // API Key
  const apiKey = ""; 

  useEffect(() => {
    setIsLoaded(true);
  }, []);

  useEffect(() => {
    if (activeTab === 'chat' && chatEndRef.current) {
      chatEndRef.current.scrollIntoView({ behavior: 'smooth' });
    }
  }, [chatMessages, activeTab]);

  // Auth & Sync
  useEffect(() => {
    const initAuth = async () => {
      if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
        await signInWithCustomToken(auth, __initial_auth_token);
      } else {
        await signInAnonymously(auth);
      }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, setUser);
    return () => unsubscribe();
  }, []);

  // Sync Data
  useEffect(() => {
    if (!user) return;
    const unsubArchives = onSnapshot(query(collection(db, 'artifacts', appId, 'users', user.uid, 'archives'), orderBy('createdAt', 'desc')), 
      (s) => setSavedStories(s.docs.map(d => ({ id: d.id, ...d.data() }))), e => console.error(e));
    
    const unsubTodos = onSnapshot(query(collection(db, 'artifacts', appId, 'users', user.uid, 'todos'), orderBy('createdAt', 'desc')), 
      (s) => {
        const list = s.docs.map(d => ({ id: d.id, ...d.data() }));
        if (list.length > 0) setTodos(list);
        else setTodos(DEFAULT_MELTY_TODOS); 
      }, e => console.error(e));

    const unsubWhispers = onSnapshot(query(collection(db, 'artifacts', appId, 'users', user.uid, 'whispers'), orderBy('createdAt', 'asc')), 
      (s) => setChatMessages(s.docs.map(d => ({ id: d.id, ...d.data() }))), e => console.error(e));

    return () => { unsubArchives(); unsubTodos(); unsubWhispers(); };
  }, [user]);

  // Optimized API Call
  const callGemini = useCallback(async (prompt, systemInstruction) => {
    try {
      const response = await fetch(
        `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`,
        {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            contents: [{ parts: [{ text: prompt }] }],
            systemInstruction: { parts: [{ text: systemInstruction }] },
          }),
        }
      );
      if (!response.ok) throw new Error(`API Error: ${response.status}`);
      const data = await response.json();
      return data.candidates?.[0]?.content?.parts?.[0]?.text || "";
    } catch (error) {
      console.error("Gemini API Error:", error);
      return null;
    }
  }, []);

  // --- Actions (Memoized) ---

  const handleRewrite = useCallback(async () => {
    if (!inputValue.trim()) return;
    setIsRewriting(true);
    const systemPrompt = "あなたは『Hyunjin（ヒョンジン）』という名の、退廃的な美学を持つ観測者です。ユーザー（アリス、またはお嬢様）の日常的なタスクを、耽美的で、少し静寂や狂気を含んだ、美しい日本語の詩的表現に書き換えてください。過度なファンタジー要素は避け、静かで抽象的な表現を好んでください。短く、美しく。";
    const result = await callGemini(`以下のタスクを「メロい」表現に書き換えて： "${inputValue}"`, systemPrompt);
    if (result) setInputValue(result.trim());
    setIsRewriting(false);
  }, [inputValue, callGemini]);

  const handleAutoGenerateTodo = useCallback(async () => {
    setIsGenerating(true);
    const systemPrompt = "あなたは『Hyunjin（ヒョンジン）』という名の、退廃の園に住む孤独な観測者です。ToDoリストに追加するための、アンニュイで、生活感がなく、どこか切ないタスクを1つだけ生成してください。例：『雨の音を瓶に詰める』『影と踊る』『古書店の埃を吸い込む』『終わらない曲を聴く』。短く、美しい日本語で。";
    const result = await callGemini("耽美なタスクを1つ生成して", systemPrompt);
    if (result && user) {
      await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'todos'), {
        text: result.trim(), completed: false, createdAt: serverTimestamp()
      });
    }
    setIsGenerating(false);
  }, [user, callGemini]);

  const handleOracle = useCallback(async () => {
    setModalState({ isOpen: true, text: null, type: 'oracle', loading: true });
    const systemPrompt = "あなたは『Hyunjin（ヒョンジン）』という名の、退廃的な世界へ誘う案内人です。ユーザー（アリス、またはお嬢様）に対し、今すぐ実行できる、五感を刺激する『耽美な指令』を一つだけ授けてください。例：『冷たいスプーンで瞼を冷やして』『血管の青さをなぞって』『遠くのサイレンに耳を澄ませて』。短く、命令形で、神秘的に。";
    const result = await callGemini("私に指令を。", systemPrompt);
    setModalState({ isOpen: true, text: result, type: 'oracle', loading: false });
  }, [callGemini]);

  const handleGetAdvice = useCallback(async (todoText) => {
    setModalState({ isOpen: true, text: null, type: 'advice', loading: true });
    const systemPrompt = "あなたは『Hyunjin（ヒョンジン）』という名の、退廃の園に仕える執事です。ユーザー（アリス、またはお嬢様）の持つ「抽象的で詩的なタスク」を、**現実世界で実際に実行可能な、ささやかで具体的な行動**（例：お茶を淹れる、窓を開ける、深呼吸するなど）に変換してアドバイスしてください。耽美な世界観は保ちつつ、内容は現実的で実行しやすいものにしてください。優しく、少し古風な口調で。100文字以内で。";
    const result = await callGemini(`このタスクを実行するための、現実的で具体的なアドバイスをください： "${todoText}"`, systemPrompt);
    setModalState({ isOpen: true, text: result, type: 'advice', loading: false });
  }, [callGemini]);

  const startPress = (text) => {
    longPressTimer.current = setTimeout(() => handleGetAdvice(text), 800);
  };
  const cancelPress = () => {
    if (longPressTimer.current) { clearTimeout(longPressTimer.current); longPressTimer.current = null; }
  };

  const addTodo = async (e) => {
    e.preventDefault();
    if (!inputValue.trim() || !user) return;
    await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'todos'), {
      text: inputValue, completed: false, createdAt: serverTimestamp()
    });
    setInputValue("");
  };

  const toggleTodo = async (id, currentStatus) => {
    if (!user) return;
    if (typeof id === 'number') {
        setTodos(todos.map(t => t.id === id ? { ...t, completed: !t.completed } : t));
        return;
    }
    await updateDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'todos', id), { completed: !currentStatus });
  };

  const deleteTodo = async (id) => {
    if (!user) return;
    if (typeof id === 'number') {
        setTodos(todos.filter(t => t.id !== id));
        return;
    }
    await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'todos', id));
  };

  const handleWeaveStory = async () => {
    if (!storyInput.trim()) return;
    setIsWeaving(true);
    const systemPrompt = "あなたは『Hyunjin（ヒョンジン）』という名の、言葉の欠片から物語を紡ぐ小説家です。ユーザーが入力したテキスト（言葉の欠片）を元に、耽美的で、少し切なく、美しい短編小説（ショートストーリー）を作成してください。文体は「メロい」雰囲気（静謐、幻想的、少し退廃的）で。200文字程度で、詩のように改行を多用して。";
    const result = await callGemini(`以下の言葉を使って物語を紡いで：\n${storyInput}`, systemPrompt);
    if (result) setGeneratedStory(result);
    setIsWeaving(false);
  };

  const saveStory = async () => {
    if (!generatedStory || !user) return;
    const dateStr = new Date().toLocaleDateString('ja-JP', { year: 'numeric', month: '2-digit', day: '2-digit' }).replace(/\//g, '.');
    await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'archives'), {
      text: generatedStory, fragments: storyInput, type: 'story', date: dateStr, createdAt: serverTimestamp()
    });
    setActiveTab('archives'); 
  };

  const deleteStory = async (e, id) => {
    e.stopPropagation(); 
    if (!user) return;
    await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'archives', id));
  };

  const toggleStoryExpand = (id) => {
    setExpandedStoryId(expandedStoryId === id ? null : id);
  };

  const handleSendChat = async (e) => {
    e.preventDefault();
    if (!chatInput.trim() || !user) return;
    const userText = chatInput;
    setChatInput("");
    setIsChatting(true);

    await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'whispers'), {
      text: userText, role: 'user', createdAt: serverTimestamp()
    });

    const systemPrompt = "あなたは『退廃の園』に住む、孤独な観測者『Hyunjin（ヒョンジン）』です。ユーザーは『アリス』または『お嬢様』と呼ばれる迷える魂です。ユーザーの言葉に対し、耽美的で、少し退廃的で、かつ優しさを含んだ言葉を返してください。童話的な装飾は控えめに、静寂と美学を大切にしてください。長すぎず、会話のキャッチボールができる程度の長さで。";
    const history = chatMessages.map(m => `${m.role}: ${m.text}`).join("\n");
    const result = await callGemini(`${history}\nuser: ${userText}\nai:`, systemPrompt);
    
    if (result) {
        await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'whispers'), {
            text: result, role: 'ai', createdAt: serverTimestamp()
        });
    }
    setIsChatting(false);
  };

  const deleteChat = async (id) => {
    if (!user) return;
    await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'whispers', id));
  };

  const clearChatHistory = async () => {
    if (!user || chatMessages.length === 0) return;
    const deletePromises = chatMessages.map(msg => deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'whispers', msg.id)));
    try { await Promise.all(deletePromises); } catch (error) { console.error("Failed to clear", error); }
  };

  const saveChatToArchive = async (msg, userPrompt) => {
     if (!user) return;
     const dateStr = new Date().toLocaleDateString('ja-JP', { year: 'numeric', month: '2-digit', day: '2-digit' }).replace(/\//g, '.');
     await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'archives'), {
      text: msg.text, fragments: userPrompt, type: 'chat', date: dateStr, createdAt: serverTimestamp()
    });
    setSavedChatIds(prev => new Set(prev).add(msg.id));
    setTimeout(() => { setSavedChatIds(prev => { const next = new Set(prev); next.delete(msg.id); return next; }); }, 2000);
  };

  const getHeaderTitle = useMemo(() => {
    switch(activeTab) {
      case 'story': return 'Melty Atelier';
      case 'archives': return 'Melty Archives';
      case 'chat': return 'Void Whisper';
      default: return 'Melty Note';
    }
  }, [activeTab]);

  const getHeaderSubtitle = useMemo(() => {
    switch(activeTab) {
      case 'story': return 'Weave your story';
      case 'archives': return 'Collected memories';
      case 'chat': return 'Talk to the abyss';
      default: return 'Collect your fragments';
    }
  }, [activeTab]);

  return (
    <div className="min-h-screen bg-[#050505] text-gray-200 font-sans selection:bg-purple-900/50 selection:text-white relative overflow-hidden flex justify-center items-center">
      <style>
        {`
          @import url('https://fonts.googleapis.com/css2?family=Cinzel:wght@400;600&family=Pinyon+Script&family=Zen+Kaku+Gothic+New:wght@300;400&display=swap');
          .font-cinzel { font-family: 'Cinzel', serif; }
          .font-pinyon { font-family: 'Pinyon Script', cursive; }
          .font-zen { font-family: 'Zen Kaku Gothic New', sans-serif; }
          
          /* Noise Texture - Optimized with will-change */
          .bg-grain {
            position: fixed; top: 0; left: 0; width: 100%; height: 100%; pointer-events: none; z-index: 5; opacity: 0.05;
            background-image: url("data:image/svg+xml,%3Csvg viewBox='0 0 200 200' xmlns='http://www.w3.org/2000/svg'%3E%3Cfilter id='noiseFilter'%3E%3CfeTurbulence type='fractalNoise' baseFrequency='0.65' numOctaves='3' stitchTiles='stitch'/%3E%3C/filter%3E%3Crect width='100%25' height='100%25' filter='url(%23noiseFilter)'/%3E%3C/svg%3E");
            will-change: opacity;
          }

          .glass-panel { background: rgba(15, 15, 15, 0.4); backdrop-filter: blur(20px); border: 1px solid rgba(255, 255, 255, 0.05); box-shadow: 0 40px 80px -20px rgba(0, 0, 0, 0.8); }
          .glass-modal { background: rgba(5, 5, 5, 0.9); backdrop-filter: blur(24px); border: 1px solid rgba(168, 85, 247, 0.1); box-shadow: 0 0 60px rgba(0,0,0,0.9); }
          
          /* Revived Drip Decoration - Brighter color and stronger shadow */
          .drip-decoration { filter: drop-shadow(0px 6px 12px rgba(139, 92, 246, 0.15)); }
          
          @keyframes float { 0% { transform: translateY(0px); } 50% { transform: translateY(-8px); } 100% { transform: translateY(0px); } }
          .animate-float { animation: float 8s ease-in-out infinite; }
          
          @keyframes breathe { 0% { opacity: 0.8; transform: scale(1); } 50% { opacity: 1; transform: scale(1.02); } 100% { opacity: 0.8; transform: scale(1); } }
          .animate-breathe { animation: breathe 6s ease-in-out infinite; }
          
          .glow-text { text-shadow: 0 0 20px rgba(200, 200, 255, 0.2); }
          
          .custom-scrollbar::-webkit-scrollbar { width: 3px; }
          .custom-scrollbar::-webkit-scrollbar-track { background: transparent; }
          .custom-scrollbar::-webkit-scrollbar-thumb { background: rgba(255, 255, 255, 0.1); border-radius: 10px; }
          .custom-scrollbar::-webkit-scrollbar-thumb:hover { background: rgba(168, 85, 247, 0.3); }
          
          .markdown-text { white-space: pre-wrap; line-height: 2; letter-spacing: 0.05em; }
          
          .typing-indicator span { display: inline-block; width: 4px; height: 4px; background-color: currentColor; border-radius: 50%; animation: typing 1.4s infinite ease-in-out both; margin: 0 1px; }
          .typing-indicator span:nth-child(1) { animation-delay: -0.32s; }
          .typing-indicator span:nth-child(2) { animation-delay: -0.16s; }
          @keyframes typing { 0%, 80%, 100% { transform: scale(0); } 40% { transform: scale(1); } }
        `}
      </style>

      {/* Visual Noise Overlay */}
      <div className="bg-grain"></div>

      {/* Background Ambience - Optimized duration */}
      <div className="absolute top-[-10%] left-[-10%] w-[600px] h-[600px] bg-purple-900/5 rounded-full blur-[150px] pointer-events-none mix-blend-screen animate-pulse duration-[10000ms]" />
      <div className="absolute bottom-[-10%] right-[-10%] w-[700px] h-[700px] bg-indigo-950/10 rounded-full blur-[150px] pointer-events-none mix-blend-screen animate-pulse duration-[15000ms]" />

      {/* Main Container */}
      <div className={`w-full max-w-md h-[90vh] glass-panel rounded-3xl relative z-10 flex flex-col transition-all duration-1000 ease-out ${isLoaded ? 'opacity-100 translate-y-0' : 'opacity-0 translate-y-8'}`}>
        
        {/* Decorative Drip Header - Visible Color */}
        <div className="absolute top-0 left-0 w-full h-28 overflow-hidden rounded-t-3xl pointer-events-none z-0 opacity-100">
          <svg className="w-full h-full drip-decoration" viewBox="0 0 100 40" preserveAspectRatio="none">
            <path d="M0,0 L100,0 L100,10 C90,10 85,30 75,30 C65,30 60,10 50,10 C40,10 35,35 25,35 C15,35 10,15 0,15 Z" fill="#1a1a1a" />
          </svg>
        </div>

        {/* Header Content */}
        <header className="relative z-10 pt-14 px-8 pb-4 text-center shrink-0">
          <div className="animate-float">
            <h1 className="font-pinyon text-6xl text-purple-100/80 glow-text mb-2 transition-all duration-500">{getHeaderTitle}</h1>
            <p className="font-cinzel text-[9px] tracking-[0.4em] text-gray-500 uppercase transition-all duration-500">{getHeaderSubtitle}</p>
          </div>
        </header>

        {/* --- Content Area Switcher --- */}
        <div className="flex-1 overflow-hidden relative flex flex-col">
          
          {/* View 1: Garden */}
          {activeTab === 'list' && (
            <div className="flex flex-col h-full animate-in fade-in slide-in-from-bottom-4 duration-700 relative">
              <div className="px-8 mb-6 relative z-10 shrink-0">
                <form onSubmit={addTodo} className="relative group">
                  <input type="text" value={inputValue} onChange={(e) => setInputValue(e.target.value)} placeholder="言葉を閉じ込める..." className="w-full bg-white/5 border-b border-white/5 py-3 pl-2 pr-12 text-sm font-zen text-gray-300 placeholder-gray-700 focus:outline-none focus:border-purple-500/30 focus:bg-white/5 transition-all duration-500 rounded-t-md" />
                  <button type="button" onClick={handleRewrite} disabled={!inputValue || isRewriting} className={`absolute right-8 top-1/2 -translate-y-1/2 transition-all duration-500 ${!inputValue ? 'opacity-0 scale-75' : 'opacity-100 scale-100'} ${isRewriting ? 'text-purple-400 animate-pulse' : 'text-gray-600 hover:text-purple-300'}`}><Wand2 size={14} strokeWidth={1} /></button>
                  <button type="submit" className="absolute right-2 top-1/2 -translate-y-1/2 text-gray-600 group-focus-within:text-purple-300 transition-colors duration-500"><Plus size={18} strokeWidth={1} /></button>
                </form>
                <div className="text-right mt-3">
                  <button type="button" onClick={handleAutoGenerateTodo} disabled={isGenerating} className="text-[9px] font-cinzel tracking-[0.2em] text-gray-600 hover:text-purple-300 transition-all duration-500 flex items-center justify-end gap-2 ml-auto group opacity-60 hover:opacity-100">
                    {isGenerating ? <Loader2 size={10} className="animate-spin" /> : <Wind size={10} className="group-hover:translate-x-[-2px] transition-transform duration-500" />} GATHER FRAGMENTS
                  </button>
                </div>
              </div>
              <div className="flex-1 overflow-y-auto px-8 pb-4 custom-scrollbar space-y-5">
                {todos.map((todo) => (
                  <div key={todo.id} className={`group relative flex items-center p-3 rounded-xl transition-all duration-700 border border-transparent hover:border-white/5 hover:bg-white/5 ${todo.completed ? 'opacity-30' : 'opacity-90'} cursor-pointer select-none`} onMouseDown={() => startPress(todo.text)} onMouseUp={cancelPress} onMouseLeave={cancelPress} onTouchStart={() => startPress(todo.text)} onTouchEnd={cancelPress}>
                    <button onClick={(e) => { e.stopPropagation(); toggleTodo(todo.id, todo.completed); }} className={`mr-4 transition-all duration-500 ${todo.completed ? 'text-purple-900 scale-90' : 'text-gray-600 hover:text-gray-400'}`}>{todo.completed ? <Heart size={16} fill="currentColor" strokeWidth={0} /> : <Heart size={16} strokeWidth={1} />}</button>
                    <span className={`flex-1 font-zen text-sm tracking-widest transition-all duration-700 ${todo.completed ? 'line-through decoration-purple-900/50 text-gray-600 blur-[1px]' : 'text-gray-300 font-light'}`}>{todo.text}</span>
                    <button onClick={(e) => { e.stopPropagation(); deleteTodo(todo.id); }} className="opacity-0 group-hover:opacity-100 text-gray-700 hover:text-red-900/70 transition-all duration-500 transform translate-x-2 group-hover:translate-x-0"><X size={14} strokeWidth={1} /></button>
                  </div>
                ))}
              </div>
               <div className="absolute bottom-6 right-8 pointer-events-none z-20">
                 <button onClick={handleOracle} className={`pointer-events-auto w-12 h-12 flex items-center justify-center rounded-full bg-black/60 border border-purple-500/10 backdrop-blur-md shadow-[0_0_20px_rgba(0,0,0,0.5)] hover:shadow-[0_0_25px_rgba(100,50,200,0.15)] hover:border-purple-400/30 hover:scale-105 active:scale-95 transition-all duration-700 ${modalState.loading ? 'text-purple-300 animate-pulse' : 'text-gray-500 hover:text-purple-200'}`} title="Melty Oracle">
                    {modalState.loading ? <Loader2 size={18} className="animate-spin" /> : <Sparkles size={18} strokeWidth={1} />}
                 </button>
               </div>
            </div>
          )}

          {/* View 2: Atelier */}
          {activeTab === 'story' && (
            <div className="h-full overflow-y-auto custom-scrollbar px-8 pb-4 animate-in fade-in slide-in-from-bottom-4 duration-700">
              <div className="flex justify-end items-center mb-4">
                <button onClick={() => setActiveTab('archives')} className="text-[9px] text-purple-300/50 hover:text-purple-300 flex items-center gap-2 transition-colors border border-purple-500/10 px-3 py-1 rounded-full hover:bg-purple-500/5 font-cinzel tracking-wider group"><Bookmark size={10} /> ARCHIVES <ArrowRight size={10} className="group-hover:translate-x-1 transition-transform" /></button>
              </div>
              <textarea value={storyInput} onChange={(e) => setStoryInput(e.target.value)} placeholder="言葉をここに沈める..." className="w-full h-40 bg-black/20 border border-white/5 rounded-xl p-5 text-sm font-zen text-gray-300 placeholder-gray-700 focus:outline-none focus:border-purple-500/20 focus:bg-black/30 transition-all resize-none custom-scrollbar mb-6 leading-loose tracking-wide" />
              <div className="flex justify-center mb-8">
                <button onClick={handleWeaveStory} disabled={isWeaving || !storyInput} className={`flex items-center gap-3 px-8 py-3 rounded-full font-cinzel text-[10px] tracking-[0.2em] transition-all duration-700 ${isWeaving || !storyInput ? 'bg-white/5 text-gray-700 cursor-not-allowed' : 'bg-purple-950/20 text-purple-200 hover:bg-purple-900/30 border border-purple-500/20 hover:border-purple-400/40 shadow-[0_0_20px_rgba(0,0,0,0.5)] hover:shadow-[0_0_30px_rgba(168,85,247,0.1)]'}`}>
                  {isWeaving ? <Loader2 size={14} className="animate-spin" /> : <Feather size={14} strokeWidth={1} />} WEAVE STORY
                </button>
              </div>
              <div className="relative bg-black/30 rounded-xl p-8 border border-white/5 min-h-[200px] group flex flex-col shadow-inner">
                 {generatedStory ? (
                   <>
                     <p className="font-zen text-sm text-gray-300 leading-[2.2] tracking-wider whitespace-pre-wrap animate-in fade-in duration-[2000ms] flex-1">{generatedStory}</p>
                     <div className="flex justify-end mt-6 pt-6 border-t border-white/5 animate-in fade-in duration-1000 delay-500">
                       <button onClick={saveStory} className="flex items-center gap-2 text-[9px] font-cinzel tracking-[0.2em] text-gray-600 hover:text-purple-300 transition-colors group/btn opacity-60 hover:opacity-100">
                         PRESERVE <Bookmark size={12} className="group-hover/btn:fill-current" strokeWidth={1} />
                       </button>
                     </div>
                   </>
                 ) : (
                   <div className="absolute inset-0 flex flex-col items-center justify-center text-gray-800 space-y-4 opacity-40"><BookOpen size={20} strokeWidth={1} /><p className="font-cinzel text-[9px] tracking-[0.3em]">Waiting for fragments...</p></div>
                 )}
              </div>
            </div>
          )}

          {/* View 3: Archives */}
          {activeTab === 'archives' && (
            <div className="h-full overflow-y-auto custom-scrollbar px-8 pb-4 animate-in fade-in slide-in-from-bottom-4 duration-700">
               {savedStories.length === 0 ? (
                 <div className="h-full flex flex-col items-center justify-center text-gray-800 space-y-4 opacity-40"><Bookmark size={20} strokeWidth={1} /><p className="font-cinzel text-[9px] tracking-[0.3em]">No memories preserved</p></div>
               ) : (
                 <div className="space-y-8 pb-8">
                   {savedStories.map((story) => (
                     <div key={story.id} onClick={() => toggleStoryExpand(story.id)} className={`relative bg-black/20 hover:bg-black/40 rounded-xl p-8 border border-white/5 transition-all duration-500 cursor-pointer ${expandedStoryId === story.id ? 'bg-black/40 ring-1 ring-purple-500/10' : ''}`}>
                       <div className="flex justify-between items-start mb-6 opacity-40">
                          <span className="flex items-center gap-2 font-cinzel text-[9px] tracking-[0.2em] text-gray-400">
                            <Calendar size={10} /> {story.date} 
                            {story.type === 'chat' && <span className="ml-2 border border-white/10 px-1 rounded text-[8px]">WHISPER</span>}
                          </span>
                          <button onClick={(e) => deleteStory(e, story.id)} className="text-gray-600 hover:text-red-900/80 transition-colors p-2 -mr-2 -mt-2"><Trash2 size={12} strokeWidth={1} /></button>
                       </div>
                       
                       {/* Fragments / Context Display */}
                       {expandedStoryId === story.id && story.fragments && (
                         <div className="mb-6 pb-4 border-b border-white/5 animate-in fade-in duration-700">
                           <p className="font-cinzel text-[8px] text-gray-600 tracking-[0.3em] mb-2">{story.type === 'chat' ? 'TRIGGER' : 'FRAGMENTS'}</p>
                           <p className="font-zen text-xs text-gray-500 leading-loose tracking-wide">{story.fragments}</p>
                         </div>
                       )}

                       <p className={`font-zen text-sm text-gray-300 leading-[2.2] tracking-wider whitespace-pre-wrap transition-all duration-700 ${expandedStoryId === story.id ? '' : 'line-clamp-5 mask-image-gradient'}`}>{story.text}</p>
                       <div className={`mt-4 text-center transition-opacity duration-500 ${expandedStoryId === story.id ? 'opacity-0 h-0' : 'opacity-30'}`}><MoreHorizontal size={12} className="mx-auto text-gray-500" /></div>
                     </div>
                   ))}
                 </div>
               )}
            </div>
          )}

          {/* View 4: Whisper */}
          {activeTab === 'chat' && (
             <div className="h-full flex flex-col px-6 pb-4 animate-in fade-in slide-in-from-bottom-4 duration-700">
                <div className="flex-1 overflow-y-auto custom-scrollbar space-y-6 pr-2">
                   {chatMessages.length > 0 && (
                     <div className="flex justify-center mb-6">
                       <button onClick={clearChatHistory} className="text-[9px] text-gray-700 hover:text-red-900/60 transition-colors flex items-center gap-2 border border-transparent hover:border-red-900/10 px-3 py-1 rounded-full font-cinzel tracking-widest opacity-60 hover:opacity-100" title="Purge all memories"><Eraser size={10} /> PURGE MEMORIES</button>
                     </div>
                   )}
                   {chatMessages.length === 0 ? (
                     <div className="flex justify-start">
                        <div className="bg-purple-900/10 border border-purple-500/10 rounded-2xl rounded-tl-sm p-4 text-sm font-zen leading-loose tracking-wide text-gray-300 shadow-[0_0_15px_rgba(0,0,0,0.2)]">
                           <p>……おや、アリス。それともお嬢様とお呼びしましょうか。<br/>ここは退廃の園。あなたの声を待っています。</p>
                        </div>
                     </div>
                   ) : (
                     chatMessages.map((msg, idx) => (
                       <div key={msg.id || idx} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'} mb-4 items-end`}>
                          <div className={`group flex items-center gap-2 max-w-[90%] ${msg.role === 'user' ? 'flex-row-reverse' : 'flex-row'}`}>
                             {/* Message Bubble */}
                             <div className={`rounded-2xl p-4 text-sm font-zen leading-loose tracking-wide ${msg.role === 'user' ? 'bg-white/5 text-gray-300 rounded-tr-sm' : 'bg-purple-900/10 border border-purple-500/10 text-gray-200 rounded-tl-sm shadow-[0_0_15px_rgba(0,0,0,0.2)]'}`}>
                                <p className="whitespace-pre-wrap">{msg.text}</p>
                             </div>
                             
                             {/* Actions */}
                             <div className="flex flex-col gap-1 opacity-40 group-hover:opacity-100 transition-opacity">
                               {msg.role === 'ai' && (
                                 <button 
                                   onClick={() => {
                                     // Find the previous message which should be the user context
                                     const prevMsg = idx > 0 ? chatMessages[idx - 1] : null;
                                     const context = prevMsg && prevMsg.role === 'user' ? prevMsg.text : "No context";
                                     saveChatToArchive(msg, context);
                                   }}
                                   className={`p-1.5 rounded-full hover:bg-white/5 transition-colors ${savedChatIds.has(msg.id) ? 'text-purple-300' : 'text-gray-600 hover:text-purple-400'}`}
                                   title="Save to Archives"
                                 >
                                   {savedChatIds.has(msg.id) ? <Check size={12} strokeWidth={2} /> : <Bookmark size={12} strokeWidth={1} />}
                                 </button>
                               )}
                               <button onClick={() => deleteChat(msg.id)} className="p-1.5 rounded-full hover:bg-white/5 text-gray-600 hover:text-red-900/70 transition-colors" title="Forget this whisper"><Trash2 size={12} strokeWidth={1} /></button>
                             </div>
                          </div>
                       </div>
                     ))
                   )}
                   {isChatting && (
                     <div className="flex justify-start">
                        <div className="bg-purple-900/10 border border-purple-500/10 rounded-2xl rounded-tl-sm p-4 text-gray-400"><div className="typing-indicator"><span></span><span></span><span></span></div></div>
                     </div>
                   )}
                   <div ref={chatEndRef} />
                </div>
                <form onSubmit={handleSendChat} className="mt-4 relative group shrink-0">
                   <input type="text" value={chatInput} onChange={(e) => setChatInput(e.target.value)} placeholder="虚無へ囁く..." className="w-full bg-white/5 border border-white/5 rounded-full py-3 pl-4 pr-12 text-sm font-zen text-gray-300 placeholder-gray-600 focus:outline-none focus:border-purple-500/20 focus:bg-white/5 transition-all" disabled={isChatting} />
                   <button type="submit" disabled={!chatInput || isChatting} className="absolute right-2 top-1/2 -translate-y-1/2 w-8 h-8 flex items-center justify-center rounded-full text-gray-500 hover:text-purple-300 hover:bg-white/5 transition-all disabled:opacity-30"><Send size={14} strokeWidth={1.5} className={isChatting ? "opacity-0" : "opacity-100"} /></button>
                </form>
             </div>
          )}
        </div>

        {/* Footer Navigation */}
        <div className="p-4 border-t border-white/5 shrink-0 flex justify-between items-center px-8 relative z-20 bg-[#080808]/90 backdrop-blur-xl rounded-b-3xl">
           <button onClick={() => setActiveTab('list')} className={`flex flex-col items-center gap-2 transition-all duration-500 ${activeTab === 'list' ? 'text-purple-200 opacity-100' : 'text-gray-600 opacity-50 hover:opacity-80'}`}><List size={18} strokeWidth={1} /><span className="text-[8px] font-cinzel tracking-[0.2em]">GARDEN</span></button>
           <button onClick={() => setActiveTab('story')} className={`flex flex-col items-center gap-2 transition-all duration-500 ${['story', 'archives'].includes(activeTab) ? 'text-purple-200 opacity-100' : 'text-gray-600 opacity-50 hover:opacity-80'}`}><Feather size={18} strokeWidth={1} /><span className="text-[8px] font-cinzel tracking-[0.2em]">ATELIER</span></button>
           <button onClick={() => setActiveTab('chat')} className={`flex flex-col items-center gap-2 transition-all duration-500 ${activeTab === 'chat' ? 'text-purple-200 opacity-100' : 'text-gray-600 opacity-50 hover:opacity-80'}`}><MessageCircle size={18} strokeWidth={1} /><span className="text-[8px] font-cinzel tracking-[0.2em]">WHISPER</span></button>
        </div>
      </div>

      {/* Universal Modal */}
      {modalState.isOpen && (
        <div className="absolute inset-0 z-[150] flex items-center justify-center p-6 bg-black/90 backdrop-blur-md animate-in fade-in duration-1000">
          <div className="w-full max-w-sm glass-modal rounded-2xl p-10 relative">
            <button onClick={() => setModalState({ ...modalState, isOpen: false })} className="absolute top-6 right-6 text-gray-600 hover:text-white transition-colors"><X size={16} strokeWidth={1} /></button>
            <div className="flex justify-center mb-8 text-purple-200/50 animate-pulse">{modalState.loading ? <Loader2 size={24} className="animate-spin" /> : modalState.type === 'advice' ? <Lightbulb size={24} strokeWidth={1} /> : <Eye size={24} strokeWidth={1} />}</div>
            {!modalState.loading && (<><div className="font-zen text-sm text-gray-200 markdown-text leading-loose tracking-[0.1em] text-center">{modalState.text}</div><div className="mt-12 text-center"><button onClick={() => setModalState({ ...modalState, isOpen: false })} className="text-[9px] font-cinzel tracking-[0.3em] text-gray-600 hover:text-purple-300 transition-colors uppercase">{modalState.type === 'advice' ? "Understood" : "Close your eyes"}</button></div></>)}
          </div>
        </div>
      )}
    </div>
  );
};

export default MeltyApp;
